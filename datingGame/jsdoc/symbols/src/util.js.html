 84 function arrayShuffle(oldArray) {
 85 	var newArray = oldArray.slice();
 86  	var len = newArray.length;
 87 	var i = len;
 88 	 while (i--) {
 89 	 	var p = parseInt(Math.random()*len);
 90 		var t = newArray[i];
 91   		newArray[i] = newArray[p];
 92 	  	newArray[p] = t;
 93  	}
 94 	return newArray; 
 95 };
